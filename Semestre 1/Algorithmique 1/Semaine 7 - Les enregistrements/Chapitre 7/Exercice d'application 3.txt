Constante 
  Nmax = 100
Type
  Domicile = Enregistrement
    Avenue, ville, pays : chaine [20]
    Numero, code_postale : entier
    Téléphone : chaine [10]
  Fin Enregistrement
  Tab = Tableau [1..Nmax] de chaine de caractère
  Client = Enregistrement
    nom : chaine [15]
    Pr : chaine [20]
    Adr: Domicile
    Num_compte : Entier
    Solde : réel
    Historique: Tab
  Fin Enregistrement
Var
  C : Client
  n, i : entier
Debut
  Ecrire("programme de gestion d'une banque")
  Ecrire('**********Information du client ******** ')
  Ecrire("entrer le nom du client :")
  Lire(C.nom)
  Ecrire("entrer le prénom du client :")
  Lire(C.Pr)
  Ecrire("entrer le numéro du compte:")
  Lire(C.num)
  Ecrire("entrer l’adresse du client :")
  Lire(C.Adr.Avenue, C.Adr.ville, C.Adr.pays, C.Adr.Numero, C.Adr.code_postale, C.Adr.Téléphone)
  Ecrire("entrer le solde du client :")
  Lire(C.solde)
  Ecrire("**********Historique transactions********")
  Répéter
    Lire (n)
  Jusqu'à (n>0) ET (n<=Nmax)
  Pour i allant de 1 à n faire
    lire (C.Historique[i])
  Finpour
  Ecrire("**********Affichage information Client********")
  Ecrire("Nom du client :", C.nom)
  Ecrire("Prénom du client :", C.Pr)
  Ecrire("Numéro du compte:", C.num)
  Ecrire(" Adresse du client :", C.Adr.Avenue,C.Adr.ville,C.Adr.pays, C.Adr.Numero, C.Adr.code_postale, C.Adr.Téléphone)
  Ecrire("***Affichage information Client: Historique ***")
  Pour i allant de 1 à n faire
    Ecrire (C.Historique[i])
  Finpour
Fin
