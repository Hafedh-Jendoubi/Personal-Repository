Tri par selection: 

#include <stdio.h>

int main()
{
    int i, j, temp, min, T[6]={60, 50, 20, 40, 10, 30};

    for(i=0; i<5; i++){
        min = i;
        for(j=i+1; j<6; j++){
            if(T[j]<T[min])
                min = j;
        }
        temp = T[i];
        T[i] = T[min];
        T[min] = temp;
    }

    for(i=0; i<6; i++)
        printf("%d | ", T[i]);

    return 0;
}
