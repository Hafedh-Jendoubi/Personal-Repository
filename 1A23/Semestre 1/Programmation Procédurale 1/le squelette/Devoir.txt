functions.c

#include "functions.h"

void menu(int *ch)
{
    if(*ch == -1){
        printf("~~~~~~~~ Menu: ~~~~~~~~\n");
        printf("0- Exit\n");
    }
    do{
        printf("\nTaper votre choix: "); scanf("%d", &*ch);
    }while((*ch<0)||(*ch>1));
}

----------------------------------------------------------

main.c

#include "functions.h"

int main()
{
    int ch=-1;

    do{
        menu(&ch);
        switch(ch){
            case 0:
                printf("...\n"); break;

            case 1:
                break;
        }
    }while(ch!=0);

    return 0;
}
